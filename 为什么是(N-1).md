$$
\displaylines{
\begin{equation*}
\begin{aligned}
S^2 &= \frac{{1}}{n-1} \sum\limits_{i=1}^{n} (x_{i} - \bar{x})^2 \\ \\
E(x_{i}) &= \mu \\
Var(x_{i}) &= \sigma^2 \\
E(\bar{x}) &= \mu \\
Var(\bar{x}) &= \frac{\sigma^2}{n} \\
Var(x_{i}) &= E(x_{i}^2) - (E(x_{i}))^2
\\
希望E(S^2) = \sigma^2。 \\
\\
E(S^2) &= E\left( \frac{1}{n-1} \sum\limits_{i=1}^{n}(x_{i} - \bar{x})^2 \right) \\
	&= \frac{1}{n-1} E\left( \sum\limits_{i=1}^{n}(x_{i} - \bar{x})^2 \right) = ……\\
\\
与此同时，\sum\limits_{i=1}^{n}(x_{i} - \bar{x})^2 &= \sum\limits_{i=1}^{n}(x_{i}^2-2x_{i}\bar{x}+\bar{x}^2) \\
	&= \sum\limits_{i=1}^{n}x_{i}^2 - 2\bar{x}\sum\limits_{i=1}^{n}x_{i} + n\bar{x}^2 \\
	& (n，\bar{x}为常数，可直接被析出。)\\
	&= \sum\limits_{i=1}^{n}x_{i}^2 - 2n\bar{x}^2 + \bar{n}x^2 \\
	& \left( \sum\limits_{i=1}^{n}x_{i} = n\bar{x} \right)\\
	&= \sum\limits_{i=1}^{n}x_{i}^2 - n\bar{x}^2\\
\\
    \frac{1}{n-1} E\left( \sum\limits_{i=1}^{n}(x_{i} - \bar{x})^2 \right) &= \frac{1}{n-1} E\left( \sum\limits_{i=1}^{n}x_{i}^2 - n\bar{x}^2 \right) \\ 
    &= \frac{1}{n-1} \left( E\left( \sum\limits_{i=1}^{n}x_{i}^2 \right) - E(n\bar{x}^2) \right) = …… \\
\\
与此同时，E\left( \sum\limits_{i=1}^{n}x_{i}^2 \right) &= E(x_{1}^2 + x_{2}^2 +……+x_{n}^2) \\
    &= E(x_{1}^2) + E(x_{2}^2) + …… + E(x_{n}^2) \\
    &= nE(x_{i}^2) = …………\\
\\
与此同时，E(x_{i}^2) &= Var(x_{i}) + E(x_{i})^2 \\
    &= \sigma^2 + \mu^2 \\
\\
    nE(x_{i}^2) &= n\sigma^2 + n\mu^2 \\
\\
与此同时，E(n\bar{x}^2) &= nE(\bar{x}^2) \\
    &= n(Var(\bar{x})+E(\bar{x})^2) \\
    &= n\left( \frac{\sigma^2}{n}+\mu^2 \right) \\
    &= \sigma^2 + n\mu^2
\\
\frac{1}{n-1} \left( E\left( \sum\limits_{i=1}^{n}x_{i}^2 \right) - E(n\bar{x}^2) \right) &= \frac{1}{n-1}((n\sigma^2+n\mu^2)-(\sigma^2 + n\mu^2)) \\
&= \frac{1}{n-1}((n-1)\sigma^2) \\
\\ \therefore E(S^2) &= \sigma^2
\end{aligned}
\end{equation*}
}
